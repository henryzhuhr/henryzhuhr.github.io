Go 1.18 еј•е…ҘдәҶеҜ№жіӣеһӢзҡ„ж”ҜжҢҒгҖӮжіӣеһӢе…Ғи®ёдҪ зј–еҶҷеҸҜд»ҘеӨ„зҗҶеӨҡз§Қж•°жҚ®зұ»еһӢзҡ„еҮҪж•°е’Ңж•°жҚ®з»“жһ„пјҢиҖҢдёҚйңҖиҰҒдёәжҜҸз§Қзұ»еһӢзј–еҶҷеҚ•зӢ¬зҡ„е®һзҺ°гҖӮ

Go жіӣеһӢзұ»еһӢжҸҗжЎҲпјҡType Parameters Proposal

е®ҳж–№ж•ҷзЁӢпјҡTutorial: Getting started with generics

вң… жҸҗзӨәпјҡGo зҡ„жіӣеһӢи®ҫи®Ўејәи°ғвҖңжңҖе°ҸеҸҜз”ЁжҖ§еҺҹеҲҷвҖқпјҢйј“еҠұејҖеҸ‘иҖ…з”ЁжңҖз®ҖеҚ•зҡ„ж–№ејҸи§ЈеҶій—®йўҳпјҢйҒҝе…ҚиҝҮеәҰеӨҚжқӮеҢ–гҖӮ

еҲӣе»әжіӣеһӢеҮҪж•° вҖӢ

жіӣеһӢеҮҪж•°зҡ„зӯҫеҗҚдёә:

go
// ж— иҝ”еӣһеҖјзҡ„жіӣеһӢеҮҪж•°
func FuncName[T any, U any](param1 T, param2 U) 

// жңүиҝ”еӣһеҖјзҡ„жіӣеһӢеҮҪж•°пјҢиҝ”еӣһзұ»еһӢдёә U
func FuncName[T any, U any](args ...T) (result U)
  • T е’Ң U жҳҜзұ»еһӢеҪўеҸӮ (type arguments)
  • any жҳҜзұ»еһӢзәҰжқҹпјҢиЎЁзӨәвҖңд»»ж„Ҹзұ»еһӢвҖқпјҢзӯүд»·дәҺ interface{}

еҲӣе»әдёҖдёӘеҸҜд»ҘеҜ№ int жҲ–иҖ… float жұӮе’Ңзҡ„еҮҪж•°пјҢдҪҝз”Ё | жқҘиЎЁзӨәиҒ”еҗҲзұ»еһӢгҖӮ

go
func SumIntsOrFloats[K comparable, V int64 | float64](m map[K]V) V {
    var s V
    for _, v := range m {
        s += v
    }
    return s
}

и°ғз”Ёиҫ“еҮәдёә

go
func TestGenerics(t *testing.T) {
    t.Logf("sum of ints : %v", SumIntsOrFloats(map[string]int64{"a": 1, "b": 2}))
    t.Logf("sum of float: %v", SumIntsOrFloats(map[string]float64{"a": 1.1, "b": 2.2}))
}
// sum of ints : 3
// sum of float: 3.3000000000000003

еҪ“з„¶д№ҹеҸҜд»Ҙе°Ҷ int64 е’Ң float64 жҠҪиұЎдёәдёҖдёӘжҺҘеҸЈпјҢжқҘе®һзҺ°жӣҙеӨҚжқӮзҡ„зұ»еһӢзәҰжқҹ (type constraint)гҖӮ

go
type Number interface {
    int64 | float64
}

з„¶еҗҺдҪҝз”ЁиҝҷдёӘжҺҘеҸЈжқҘзәҰжқҹзұ»еһӢеҸӮж•°пјҢеҪ“зұ»еһӢеҸӮж•° V ж»Ўи¶і Number жҺҘеҸЈж—¶пјҢе°ұеҸҜд»Ҙи°ғз”ЁиҝҷдёӘеҮҪж•°гҖӮ

go
func SumNumbers[K comparable, V Number](m map[K]V) V {
    var s V
    for _, v := range m {
        s += v
    }
    return s
}

зұ»еһӢеҪўеҸӮ (type arguments) вҖӢ

зұ»еһӢеҪўеҸӮ (type arguments) жҳҜжҢҮеңЁеҮҪж•°жҲ–зұ»еһӢе®ҡд№үдёӯдҪҝз”Ёзҡ„зұ»еһӢеҸӮж•°гҖӮе®ғд»¬е…Ғи®ёдҪ еңЁзј–еҶҷд»Јз Ғж—¶дёҚжҢҮе®ҡе…·дҪ“зҡ„зұ»еһӢпјҢиҖҢжҳҜеңЁи°ғз”Ёж—¶дј е…Ҙе…·дҪ“зҡ„зұ»еһӢгҖӮ

зұ»еһӢеҸӮж•°зҡ„зәҰжқҹ вҖӢ

1. иҒ”еҗҲзәҰжқҹпјҲUnion Constraintsпјү вҖӢ

дҪҝз”Ё | иЎЁзӨәеӨҡдёӘзұ»еһӢдёӯзҡ„д»»ж„ҸдёҖдёӘпјҡ

go
type Number interface {
    int | int32 | float64
}

иЎЁзӨәзұ»еһӢеҝ…йЎ»жҳҜ intгҖҒint32 жҲ– float64 дёӯзҡ„дёҖз§ҚгҖӮ

2. дәӨйӣҶзәҰжқҹпјҲIntersection Constraintsпјү вҖӢ

дҪҝз”Ё & иЎЁзӨәеӨҡдёӘжҺҘеҸЈзҡ„дәӨйӣҶпјҡ

go
type Stringer interface {
    String() string
}

type MyConstraint interface {
    comparable & Stringer
}

иЎЁзӨәзұ»еһӢеҝ…йЎ»еҗҢж—¶ж»Ўи¶і comparable е’Ң Stringer жҺҘеҸЈгҖӮ

3. з»„еҗҲдҪҝз”Ё вҖӢ

еҸҜд»Ҙж··еҗҲдҪҝз”Ё | е’Ң &пјҢдҪҶиҜ·жіЁж„Ҹдјҳе…Ҳзә§пјҡ| зҡ„дјҳе…Ҳзә§й«ҳдәҺ &гҖӮ

е»әи®®дҪҝз”ЁжӢ¬еҸ·жҳҺзЎ®дјҳе…Ҳзә§пјҡ

go
func Equals[T (int | float64) & comparable](a, b T) bool {
    return a == b
}

еёёи§ҒжіӣеһӢзұ»еһӢ вҖӢ

еҲҮзүҮзұ»еһӢ вҖӢ

еҸҜд»ҘдҪҝз”Ё []T жқҘе®ҡд№үдёҖдёӘйҖҡз”ЁеҲҮзүҮзұ»еһӢпјҡ

go
// type StringSlice []string
// type Float32Slie []float32
// type Float64Slice []float64
type Slice[T int|float32|float64 ] []T

Slice[T] жқҘиЎЁзӨәдёҖдёӘеҲҮзүҮзұ»еһӢзҡ„зұ»еһӢеҪўеҸӮгҖӮ

go
var intSlice Slice[int] = []int{1, 2, 3}  
var f32Slice Slice[float32] = []float32{1.1, 2.2, 3.3}
var f64Slice Slice[float64] = []float64{1.1, 2.2, 3.3}

еҪ“з„¶пјҢд№ҹеҸҜд»ҘдҪҝз”Ё Number жҺҘеҸЈжқҘиЎЁзӨәдёҖдёӘж•°еӯ—зұ»еһӢзҡ„зұ»еһӢеҪўеҸӮпјҢиҝҷж ·д№ҹи®ёжӣҙеҠ з¬ҰеҗҲиҜӯд№үгҖӮ

go
type Number interface {
    int | float32 | float64
}
type NumberSlice[T Number] []T

з»“жһ„дҪ“зұ»еһӢеҸӮж•° вҖӢ

еҜ№дәҺз»“жһ„дҪ“зұ»еһӢпјҢеҸҜд»ҘдҪҝз”Ё struct{...} жқҘиЎЁзӨәз»“жһ„дҪ“зҡ„зұ»еһӢеҪўеҸӮгҖӮ

go
type Pair[T, U any] struct {
    First  T
    Second U
}

дҪҝз”Ё Pair[T, U] жқҘиЎЁзӨәдёҖдёӘз»“жһ„дҪ“зҡ„зұ»еһӢеҪўеҸӮгҖӮ

go
var p1 Pair[int, string] = Pair[int, string]{First: 1, Second: "one"}
var p2 Pair[string, int] = Pair[string, int]{First: "two", Second: 2}

map зұ»еһӢ вҖӢ

еҜ№дәҺ map зұ»еһӢпјҢеҸҜд»ҘдҪҝз”Ё map[K]V жқҘиЎЁзӨәжҳ е°„зҡ„зұ»еһӢеҪўеҸӮгҖӮ

go
type Map[K comparable, V any] map[K]V

еҸҜжҜ”иҫғзұ»еһӢ вҖӢ

еңЁ Go жіӣеһӢдёӯпјҢвҖңеҸҜжҜ”иҫғзұ»еһӢвҖқ жҳҜдёҖдёӘйқһеёёеёёи§ҒгҖҒд№ҹйқһеёёйҮҚиҰҒзҡ„жҰӮеҝөгҖӮе®ғйҖҡеёёз”ЁдәҺеҲӨж–ӯдёӨдёӘеҖјжҳҜеҗҰзӣёзӯүпјҲ== е’Ң != иҝҗз®—з¬ҰпјүпјҢжҜ”еҰӮеңЁжіӣеһӢж•°жҚ®з»“жһ„дёӯжҹҘжүҫе…ғзҙ гҖҒеҺ»йҮҚгҖҒдҪңдёә map зҡ„ key зӯүж“ҚдҪңгҖӮ

д»Һ Go 1.18 ејҖе§ӢпјҢж ҮеҮҶеә“дёӯе®ҡд№үдәҶдёҖдёӘйў„еЈ°жҳҺзҡ„зәҰжқҹжҺҘеҸЈпјҡ

go
type comparable interface{ comparable }
  • comparable жҳҜдёҖдёӘжҺҘеҸЈпјҢдёҚжҳҜе…ій”®еӯ—
  • comparable еҸӘиғҪз”ЁдәҺжіӣеһӢзәҰжқҹпјҢдёҚиғҪзӣҙжҺҘиөӢеҖјз»ҷеҸҳйҮҸ
  • comparable дёҚж„Ҹе‘ізқҖж”ҜжҢҒ <гҖҒ> зӯүиҝҗз®—з¬Ұ пјҢжҜ”иҫғеӨ§е°ҸеҸҜд»ҘдҪҝз”Ёж ҮеҮҶеә“дёӯзҡ„ constraints.Ordered

иҝҷдёӘжҺҘеҸЈиЎЁзӨәвҖңиҜҘзұ»еһӢеҸҜд»Ҙиў«жҜ”иҫғвҖқпјҢдҪ еҸҜд»ҘеғҸиҝҷж ·дҪҝз”Ёе®ғпјҡ

go
func Contains[T comparable](s []T, v T) bool {
    for _, e := range s {
        if e == v {
            return true
        }
    }
    return false
}

еңЁ Go дёӯпјҢе№¶дёҚжҳҜжүҖжңүзұ»еһӢйғҪж”ҜжҢҒзӣҙжҺҘдҪҝз”Ё == жҲ– != жҜ”иҫғгҖӮд»ҘдёӢжҳҜеҸҜд»ҘжҜ”иҫғзҡ„зұ»еһӢпјҡ

зұ»еһӢжҳҜеҗҰеҸҜжҜ”иҫғ
еҹәжң¬зұ»еһӢпјҲеҰӮ int, float64, string, boolпјүвң…
жҢҮй’ҲпјҲpointerпјүвң…
жҺҘеҸЈпјҲinterfaceпјүвң…пјҲеҸӘиҰҒеҠЁжҖҒзұ»еһӢжҳҜеҸҜжҜ”иҫғзҡ„пјү
ж•°з»„пјҲarrayпјүвң…пјҲе…ғзҙ зұ»еһӢеҝ…йЎ»жҳҜеҸҜжҜ”иҫғзҡ„пјү
з»“жһ„дҪ“пјҲstructпјүвң…пјҲжүҖжңүеӯ—ж®өеҝ…йЎ»жҳҜеҸҜжҜ”иҫғзҡ„пјү
еҲҮзүҮпјҲsliceпјүгҖҒmapгҖҒеҮҪж•°вқҢ дёҚеҸҜжҜ”иҫғ

иҮӘе®ҡд№үеҸҜжҜ”иҫғзәҰжқҹ вҖӢ

дҪ д№ҹеҸҜд»ҘиҮӘе·ұе®ҡд№үеҢ…еҗ« comparable зәҰжқҹзҡ„жҺҘеҸЈпјҢжқҘз»„еҗҲе…¶д»–йҷҗеҲ¶пјҡ

go
type MyConstraint interface {
    comparable
    // е…¶д»–ж–№жі•жҲ–иҰҒжұӮ
}

жҲ–иҖ…з»“еҗҲе…·дҪ“зұ»еһӢпјҡ

go
type NumberOrString interface {
    int | string | float64
}

func FindIndex[T NumberOrString & comparable](arr []T, target T) int {
    for i, v := range arr {
        if v == target {
            return i
        }
    }
    return -1
}

еә•еұӮзұ»еһӢеҢ№й…Қ вҖӢ

Go жҳҜйқҷжҖҒзұ»еһӢиҜӯиЁҖпјҢеҜ№зұ»еһӢзҡ„еҲӨж–ӯйқһеёёдёҘж јпјҢеҚідҪҝдёӨдёӘзұ»еһӢеә•еұӮдёҖиҮҙпјҢе®ғд»¬д№ҹиў«и®ӨдёәжҳҜдёҚеҗҢзҡ„зұ»еһӢпјҡ

go
type MyInt int

var a int = 1
var b MyInt = 1
// a == b // зј–иҜ‘й”ҷиҜҜпјҡзұ»еһӢдёҚеҢ№й…Қ

дҪҶеңЁжіӣеһӢзәҰжқҹдёӯпјҢеҸҜд»ҘдҪҝз”Ё ~ иЎЁиҫҫеә•еұӮзұ»еһӢеҢ№й…Қ (underlying type matching)пјҡ

go
type Number interface {
    ~int | ~float64
}

func SomeFunc[T Number](v T) {}
SomeFunc(MyInt(1)) // вң… еҸҜд»Ҙзј–иҜ‘йҖҡиҝҮ

~ иҜӯжі•еҸӘиғҪз”ЁеңЁзұ»еһӢзәҰжқҹдёӯпјҢдёҚиғҪз”ЁдәҺеҸҳйҮҸеЈ°жҳҺпјҢдёҚиғҪз”ЁдәҺеҮҪж•°еҸӮж•°жҲ–иҝ”еӣһеҖјзұ»еһӢгҖӮ

go
var num Number = 1 // cannot use type Number outside a type constraint: interface contains type constraints

ж”ҜжҢҒеҸҜжҺ’еәҸзҡ„зәҰжқҹ вҖӢ

Go ж ҮеҮҶеә“иҝҳжҸҗдҫӣдәҶдёҖдёӘеёёз”Ёзҡ„зәҰжқҹпјҡ

go
constraints.Ordered

е®ғеҢ…еҗ«жүҖжңүж”ҜжҢҒ <гҖҒ> зӯүжҜ”иҫғиҝҗз®—зҡ„зұ»еһӢпјҲеҢ…жӢ¬ж•°еӯ—е’Ңеӯ—з¬ҰдёІпјүгҖӮ

дҪҝз”ЁиҝҷдёӘзәҰжқҹеҸҜд»Ҙе®һзҺ°жіӣеһӢжҺ’еәҸеҠҹиғҪпјҡ

go
import "golang.org/x/exp/constraints"

func Max[T constraints.Ordered](a, b T) T {
    if a > b {
        return a
    }
    return b
}
